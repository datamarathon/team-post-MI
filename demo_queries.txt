--additives, using two D tables
select d_meditems.label as mlabel, d_ioitems.label as ilabel, additives.*
from additives, d_meditems, d_ioitems
where additives.itemid = d_meditems.itemid and additives.ioitemid = d_ioitems.itemid and rownum <=10



--high troponins
select d_labitems.test_name, labevents.itemid, labevents.subject_id, labevents.hadm_id, labevents.icustay_id, labevents.charttime, labevents.valuenum
from labevents, d_labitems
where labevents.itemid in (50189, 50188) and labevents.valuenum > 1 and labevents.itemid=d_labitems.itemid and rownum <=10
