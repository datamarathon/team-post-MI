select d_chartitems.label, additives.*
from additives, d_chartitems
where additives.itemid = d_chartitems.itemid


select d_meditems.label as mlabel, d_ioitems.label as ilabel, additives.*
from additives, d_meditems, d_ioitems
where additives.itemid = d_meditems.itemid and additives.ioitemid = d_ioitems.itemid
